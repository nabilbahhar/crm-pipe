Sales CRM schema with stage-driven probabilities and RLS 

create extension if not exists pgcrypto;

create table if not exists stage_definitions (
  stage text primary key,
  sort_order int not null,
  default_probability int not null check (default_probability between 0 and 100)
);

insert into stage_definitions(stage, sort_order, default_probability) values
('Lead', 10, 10),
('Discovery', 20, 20),
('Qualified', 30, 35),
('Solutioning', 40, 50),
('Proposal Sent', 50, 60),
('Negotiation', 60, 75),
('Commit', 70, 90),
('Won', 80, 100),
('Lost / No decision', 90, 0)
on conflict (stage) do nothing;

create table if not exists accounts (
  id uuid primary key default gen_random_uuid(),
  name text not null unique,
  region text,
  segment text,
  owner_id uuid,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists contacts (
  id uuid primary key default gen_random_uuid(),
  account_id uuid not null references accounts(id) on delete cascade,
  full_name text not null,
  role text,
  email text,
  phone text,
  influence_level int check (influence_level between 1 and 5),
  notes text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists opportunities (
  id uuid primary key default gen_random_uuid(),
  account_id uuid not null references accounts(id) on delete restrict,

  title text not null,
  bu text not null check (bu in ('INFRA','CSG','CYBER','SERVICE')),
  deal_type text not null check (deal_type in ('AO','Run','Projet','Renouvellement')),

  stage text not null references stage_definitions(stage),
  probability int not null default 0 check (probability between 0 and 100),

  amount_mad numeric(14,2) not null default 0,
  margin_pct numeric(5,2) not null default 0,
  margin_mad numeric(14,2) generated always as (round(amount_mad * (margin_pct/100.0), 2)) stored,

  close_date date,
  next_step text,
  next_step_due date,

  procurement_path text,
  competition text,
  risks jsonb default '[]'::jsonb,

  owner_id uuid,
  inside_owner_id uuid,
  inside_status text not null default 'NEW'
    check (inside_status in ('NEW','IN PROGRESS','WAITING VENDOR','PRICING RECEIVED','READY TO SEND','BOOKING PENDING','DONE','BLOCKED')),

  sponsor_contact_id uuid references contacts(id),

  last_activity_at timestamptz,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),

  constraint next_step_required
    check (
      stage not in ('Proposal Sent','Negotiation','Commit')
      or next_step_due is not null
    )
);

create table if not exists activities (
  id uuid primary key default gen_random_uuid(),
  opportunity_id uuid not null references opportunities(id) on delete cascade,
  activity_type text not null check (activity_type in ('Call','Meeting','Email','Task','Other')),
  subject text not null,
  due_at timestamptz,
  done boolean not null default false,
  notes text,
  created_by uuid,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists inside_tickets (
  id uuid primary key default gen_random_uuid(),
  opportunity_id uuid not null references opportunities(id) on delete cascade,
  priority text not null default 'P2' check (priority in ('P1','P2','P3')),
  vendor text,
  request_summary text not null,
  status text not null default 'NEW'
    check (status in ('NEW','IN PROGRESS','WAITING VENDOR','PRICING RECEIVED','READY TO SEND','BOOKING PENDING','DONE','BLOCKED')),
  deadline date,
  last_update text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create or replace function set_updated_at()
returns trigger language plpgsql as $$
begin
  new.updated_at = now();
  return new;
end; $$;

create or replace function set_probability_from_stage()
returns trigger language plpgsql as $$
declare p int;
begin
  select default_probability into p from stage_definitions where stage = new.stage;
  if p is not null then new.probability = p; end if;
  return new;
end; $$;

  FROM accounts

    WHEN 0 THEN 'AO'

    WHEN 2 THEN 40
    WHEN 3 THEN 60
    WHEN 4 THEN 80

  'NONE'
FROM gen;

Populate opportunity amounts from bu_lines totals


with sums as (
  select
    o.id,
    coalesce(sum( (e->>'amount')::numeric ), 0) as sum_amount
  from public.opportunities o
  cross join lateral jsonb_array_elements(coalesce(o.bu_lines, '[]'::jsonb)) e
  group by o.id
)
update public.opportunities o
set amount = s.sum_amount
from sums s
where o.id = s.id
  and (o.amount is null or o.amount = 0)
  and s.sum_amount > 0;

Q1 2026 opportunities by status



-- Q1 2026 : somme montants OPEN + count
with o as (
  select
    id,
    status,
    bu,
    stage,
    vendor,
    amount,
    prob,
    case
      when booking_month is null then null
      when booking_month::text ~ '^\d{4}-\d{2}$' then to_date(booking_month::text||'-01','YYYY-MM-DD')
      else (booking_month::text)::date
    end as closing_date
  from public.opportunities
)
select
  status,
  count(*) as nb,
  coalesce(sum(amount),0) as total_amount
from o
where closing_date >= date '2026-01-01'
  and closing_date <  date '2026-04-01'
group by status
order by status;
